\section{Additional Exercises}

\subsection{Restricting The Atlas}

\begin{exercise}\label{ex:diffmfds-1}
Let $(\R,\cO_{\text{standard}})$ be a topological space. Let it be further equipped with an atlas $\cA = \{ (\R,x), (\R,y)\}$ where $x:\R\to\R$; $a\mapsto x(a) = a$ and $y:\R\to\R$; $a\mapsto y(a)= a^3$.

Construct the chart transition map $y\circ x^{-1}:\R \to \R$ and give its differentiability class.
\end{exercise}
\hyperref[sol:diffmfds-1]{\small \textit{Solution on p.~\pageref*{sol:diffmfds-1}}}

\begin{exercise}\label{ex:diffmfds-2}
Also construct the chart transition map $x\circ y^{-1}:\R\to\R$. Is $(\R,\cO_{\text{standard}},\cA)$ a differentiable manifold?
\end{exercise}
\hyperref[sol:diffmfds-2]{\small \textit{Solution on p.~\pageref*{sol:diffmfds-2}}}

\begin{exercise}\label{ex:diffmfds-3}
Restrict the atlas $\cA$ to an atlas $\widetilde{\cA}$ in order to make $(\R,\cO_{\text{standard}},\widetilde{\cA})$ a smooth manifold.
\end{exercise}
\hyperref[sol:diffmfds-3]{\small \textit{Solution on p.~\pageref*{sol:diffmfds-3}}}

\subsection{\texorpdfstring{Soft Squares on $\R\times\R$}{Soft Squares on R x R}}

\begin{exercise}\label{ex:diffmfds-4}
Let $\cM=\R\times\R$ equipped with the soft square topology $\cO_{ssq}$ and an atlas $\cA=\{(U_n,x_n)\}$, where $U_n = \{(x,y)\in\R\times\R \, | \, |x|<n, |y|<n, n\in \mathbb{N}^+\}$ and
\bse
    x_n : U_n \to x_n(U_n) \ss \R^2; \, (x,y) \mapsto x_n\big((x,y)\big) := \bigg(\frac{x+y}{2n}, \frac{x-y}{2n}\bigg).
\ese
Recall the definition of a chart and show that the $(U_n,x_n)$ are indeed charts.
\end{exercise}
\hyperref[sol:diffmfds-4]{\small \textit{Solution on p.~\pageref*{sol:diffmfds-4}}}

\begin{exercise}\label{ex:diffmfds-5}
Show that $\cA$ is a $C^k$-atlas by explicitly constructing the chart transition maps. What is $k$?
\end{exercise}
\hyperref[sol:diffmfds-5]{\small \textit{Solution on p.~\pageref*{sol:diffmfds-5}}}

\begin{exercise}\label{ex:diffmfds-6}
Construct at least one other chart that would lie in the maximal extension of $\cA$ and prove that it does.
\end{exercise}
