\subsection{Before The Invention Of The Wheel}

\begin{exercise}\label{ex:topologicalmfds-2}
Consider the set $F^1:= \{(m,n)\in\R^2 \, | \, m^4 + n^4 = 1\}$ of pairs of real numbers $(m,n)$. Let it be equipped with the topology subset topology $\cO_{s}|_{F^1}$ inherited from the standard topology on $\R^2$.

We look at a map $x:F^1\to \R$ that maps a pair in $F^1$ to the first entry in the pair. Write this in formal mathematical terms! Is the map injective?
\end{exercise}
\hyperref[sol:topologicalmfds-2]{\small \textit{Solution on p.~\pageref*{sol:topologicalmfds-2}}}

\begin{exercise}\label{ex:topologicalmfds-3}
This map may be made injective by restricting its domain to either of two maximal open subsets of $F^1$. Which ones? Call them $x_{\uparrow}$ and $x_{\downarrow}$.
\end{exercise}
\hyperref[sol:topologicalmfds-3]{\small \textit{Solution on p.~\pageref*{sol:topologicalmfds-3}}}

\begin{exercise}\label{ex:topologicalmfds-4}
Now, construct an injective map $y_{\uparrow}:F^1 \to \R$ that maps every pair in a maximal open subset of $F^1$ to the second entry of the pair.
\end{exercise}
\hyperref[sol:topologicalmfds-4]{\small \textit{Solution on p.~\pageref*{sol:topologicalmfds-4}}}

\begin{exercise}\label{ex:topologicalmfds-5}
Is $y_{\uparrow}$ invertible? If so, construct $y_{\uparrow}^{-1}$!
\end{exercise}
\hyperref[sol:topologicalmfds-5]{\small \textit{Solution on p.~\pageref*{sol:topologicalmfds-5}}}

\begin{exercise}\label{ex:topologicalmfds-6}
Do the domains of the maps $x_{\uparrow}$ and $y_{\uparrow}$ overlap? If so, construct the \textit{transition map} $x_{\uparrow}\circ y_{\uparrow}^{-1}$ and specify its domain and target.
\end{exercise}
\hyperref[sol:topologicalmfds-6]{\small \textit{Solution on p.~\pageref*{sol:topologicalmfds-6}}}

\begin{exercise}\label{ex:topologicalmfds-7}
How many maps (constructed this way) do you need for their domains to cover the whole of $F^1$? Does the collection of these domains and maps form an atlas on $F^1$?
\end{exercise}
