\subsection{Killing's Equation}

\begin{exercise}\label{ex:cosmology-1}
Show that a vector field $K$ is Killing if, and only if,
\bse
    (\nabla_aK)_b + (\nabla_bK)_a = 0.
\ese
\end{exercise}

\subsection{Age Of The Universe...}

\begin{exercise}\label{ex:cosmology-2}
The energy-momentum tensor for a perfect fluid is
\bse
    T^{ab} := \big[\rho(t)+p(t)\big]u^au^b + g^{ab}p(t),
\ese
where $u^a=(1,0,0,0)^a$ are the components functions of a smooth vector field and $g_{ab}$ are those of a FRW metric w.r.t. the coordinate chart $(t,r,\vartheta,\phi)$ employed in the lectures.

Derive the conservation equation
\bse
    \dot{\rho}(t) = -3\frac{\dot{a}}{a}\big(\rho(t)+p(t)\big)
\ese
by evaluating the condition
\bse
    \big(\nabla_aT\big)^{ab}u_b = 0,
\ese
which follows from the Einstein equations by virtue of the differential Bianchi identity.
\end{exercise}
\hyperref[sol:cosmology-2]{\small \textit{Solution on p.~\pageref*{sol:cosmology-2}}}

\begin{exercise}\label{ex:cosmology-3}
For $p(t)=\omega\rho(t)$, solve the conservation equation above for $\rho$ and use the Friedmann equation with vanishing spatial curvature
\bse
    \bigg(\frac{\dot{a}}{a}\bigg)^2 = \frac{8\pi G}{3}\rho,
\ese
to derive an autonomous differential equation for $a$.
\end{exercise}
\hyperref[sol:cosmology-3]{\small \textit{Solution on p.~\pageref*{sol:cosmology-3}}}

\begin{exercise}\label{ex:cosmology-4}
Show that
\bse
    a(t) = C\cdot t^{\a}, \qquad C=\text{const}
\ese
solves the autonomous differential equation equation for the scale factor $a$ if $\omega\neq 1$ and a suitably chosen $\a$.
\end{exercise}
\hyperref[sol:cosmology-4]{\small \textit{Solution on p.~\pageref*{sol:cosmology-4}}}

\begin{exercise}\label{ex:cosmology-5}
Use the result of the previous question to write down an equation for $H(t) := \frac{\dot{a}}{a}$ and estimate the age of the universe only filled with dust for today's value of the Hubble constant being given by $\frac{1}{H_0} \approx 13\times 10^9yrs$. Repeat the calculation for a universe containing only radiation.
\end{exercise}
