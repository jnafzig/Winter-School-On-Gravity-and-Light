Does $\cO_1 := \big\{ \emptyset, \{1\},\{1,2,3,4\}\big\}$ constitute a topology on $\cM$?

\subsection{Continuous Functions}

\begin{exercise}\label{ex:topology-3}
Let $\cM = \{1,2,3,4\}$ and consider the identity map $\id_{\cM}:\cM\to\cM$ defined by
\bse
    \id_{\cM}(1) = 1, \quad \id_{\cM}(2) = 2, \quad \id_{\cM}(3) = 3, \quad \id_{\cM}(4) = 4.
\ese
Is the map $\id_{\cM}$ continuous if the domain is equipped with the chaotic topology and the target with the topology $\cO_{\text{target}} := \big\{ \emptyset,\{1\},\{1,2,3,4\}\big\}$?
\end{exercise}
\hyperref[sol:topology-3]{\small \textit{Solution on p.~\pageref*{sol:topology-3}}}

\begin{exercise}\label{ex:topology-4}
Consider the inverse $\id^{-1}_{\cM}:\cM\to\cM$ of the identity map $\id_{\cM}$, such that now the target is equipped with the chaotic topology and the domain with the topology $\big\{\emptyset,\{1\},\{1,2,3,4\}\big\}$.

Provide the values of the map $\id^{-1}_{\cM}$ and decide whether $\id^{-1}_{\cM}$ is continuous!
\end{exercise}
\hyperref[sol:topology-4]{\small \textit{Solution on p.~\pageref*{sol:topology-4}}}
